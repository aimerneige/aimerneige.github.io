#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>

#define ZERO 0
#define HALF 2500000
#define MAX_SIZE 5000000

typedef struct
{
    int first;
    int last;
    int id;
} MY_ARGS;

int *arr;
int result[2];

void *myfunc(void *args)
{
    MY_ARGS *arg = (MY_ARGS *)args;
    for (int i = arg->first; i < arg->last; i++)
    {
        result[arg->id] += arr[i];
    }
    return NULL;
}

int main()
{
    arr = (int *)malloc(sizeof(int) * MAX_SIZE);
    for (int i = 0; i < MAX_SIZE; i++)
    {
        arr[i] = rand() % 5;
    }

    result[0] = 0;
    result[1] = 0;

    pthread_t th1;
    pthread_t th2;

    MY_ARGS args_1 = {ZERO, HALF, 0};
    MY_ARGS args_2 = {HALF, MAX_SIZE, 1};

    pthread_create(&th1, NULL, myfunc, &args_1);
    pthread_create(&th2, NULL, myfunc, &args_2);

    pthread_join(th1, NULL);
    pthread_join(th2, NULL);

    int S = result[0] + result[1];
    printf("Result: %d\n", S);

    return 0;
}
